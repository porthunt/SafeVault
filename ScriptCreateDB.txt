CREATE TABLE user (nome varchar[100], login char[10] PRIMARY KEY, senha char[164], grupo char[10]);
CREATE TABLE log (id INTEGER, data DATE, user char[10], FOREIGN KEY(user) REFERENCES user(id));
CREATE TABLE logDescription (id INTEGER PRIMARY KEY, descricao TEXT, FOREIGN KEY(id) REFERENCES log(id));
CREATE TABLE tries (login char[20] PRIMARY KEY, tries INTEGER, FOREIGN KEY(login) REFERENCES user(login));
CREATE TABLE randomNumbers (id char[10] PRIMARY KEY, number char[15], FOREIGN KEY(id) REFERENCES user(login));
